
-- =====================================
-- SQL JOINs
-- Sample Data
-- =====================================

DROP TABLE IF EXISTS employees;
DROP TABLE IF EXISTS departments;

CREATE TABLE departments (
    id INT PRIMARY KEY,
    department_name VARCHAR(50)
);

CREATE TABLE employees (
    id INT PRIMARY KEY,
    employee_name VARCHAR(50),
    department_id INT
);

SELECT * FROM departments;

SELECT * FROM employees;

-- INNER JOIN
SELECT
    e.employee_name,
    d.department_name
FROM employees e
INNER JOIN departments d
ON e.department_id = d.id;

-- LEFT JOIN
SELECT
    e.employee_name,
    d.department_name
FROM employees e
INNER JOIN departments d
ON e.department_id = d.id;

-- RIGHT JOIN
SELECT
    e.employee_name,
    d.department_name
FROM employees e
RIGHT JOIN departments d
ON e.department_id = d.id;

-- FULL JOIN
SELECT
    e.employee_name,
    d.department_name
FROM employees e
FULL JOIN departments d
ON e.department_id = d.id;
